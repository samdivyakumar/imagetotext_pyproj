# Zoom Registration Dashboard with Image2Text
# Put this in: /etc/nginx/sites-available/zoom-dashboard

server {
    listen 80;
    server_name 3.7.243.78;

    # Increase upload size for Image2Text app
    client_max_body_size 50M;

    # Main Zoom Dashboard (PM2 app on port 3000)
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 86400;
    }

    # Image2Text Application (Gunicorn on port 8001)
    location /image2text/ {
        proxy_pass http://127.0.0.1:8001/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Prefix /image2text;
        proxy_connect_timeout 1800;
        proxy_send_timeout 1800;
        proxy_read_timeout 1800;
    }
    
    # Redirect /image2text to /image2text/ (with trailing slash)
    location = /image2text {
        return 301 /image2text/;
    }
}
